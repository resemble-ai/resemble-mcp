Install the package:

```bash
pip install resemble
```

Configure the SDK:

```python
from resemble import Resemble

Resemble.api_key("YOUR_API_TOKEN")
```

## Projects

### List All Projects

```python
page = 1
page_size = 10

response = Resemble.v2.projects.all(page, page_size)
projects = response['items']
```

### Get a Project

```python
project_uuid = '<project_uuid>'

response = Resemble.v2.projects.get(project_uuid)
project = response['item']
```

### Create a Project

```python
name = 'My project'
description = 'My description'
is_archived = False
is_collaborative = False

response = Resemble.v2.projects.create(name, description, is_collaborative, is_archived)
project = response['item']
```

### Update a Project

```python
project_uuid = '<project_uuid>'
name = 'My project (archived)'
description = 'My description (this project has been archived)'
is_archived = True
is_collaborative = False

response = Resemble.v2.projects.update(project_uuid, name, description, is_collaborative, is_archived)
project = response['item']
```

### Delete a Project

```python
project_uuid = '<project_uuid>'

response = Resemble.v2.projects.delete(project_uuid)
```

### Create a Clip (Synchronous)

```python
response = Resemble.v2.clips.create_sync(
    project_uuid,
    voice_uuid,
    "Welcome to Resemble",
    title="Intro"
)

if response["success"]:
    clip = response["item"]
    print(clip["audio_src"])
else:
    print(response)
```

### Create a Clip (Direct Synthesis)

```python
voice_uuid = '<voice_uuid>'
project_uuid = '<project_uuid>'
output_format = 'wav'
data = 'Hello, how are you?'
title = 'Greeting'
precision = 'PCM_32'
sample_rate = 48000

response = Resemble.v2.clips.create_direct(
    project_uuid,
    voice_uuid,
    data,
    title,
    precision,
    output_format,
    sample_rate
)

clip = response['audio_content']
```

### Stream a Clip

```python
# Configure synthesis URL for streaming
Resemble.api_key('YOUR_API_TOKEN')
Resemble.syn_server_url('<syn_server_url>')  # Obtain from https://app.resemble.ai/account/api

project_uuid = '<project_uuid>'
voice_uuid = '<voice_uuid>'
body = 'This is a streaming test'

try:
    for chunk in Resemble.v2.clips.stream(project_uuid, voice_uuid, body):
        # chunk is a byte array of shorts (int 16) representing WAV audio
        # Process the audio chunk here
        pass
except Exception as e:
    # Handle exceptions
    print(f"Error: {e}")
```

### Update a Clip (Async)

```python
project_uuid = '<project_uuid>'
clip_uuid = '<clip_uuid>'
voice_uuid = '<voice_uuid>'

body = 'This is an updated async test'

response = Resemble.v2.clips.update_async(
    project_uuid,
    clip_uuid,
    voice_uuid,
    callback_uri,
    body,
    title=None,
    sample_rate=None,
    output_format=None,
    precision=None,
    include_timestamps=None,
    is_archived=None
)
clip = response['item']
```

## Deepfake Detection

### Create a Detection

```python
media_url = "https://example.com/sample-video.mp4"

response = Resemble.v2.deepfake_detection.detect(url=media_url)

if response["success"]:
    detection_uuid = response["item"]["uuid"]
    status = response["item"]["status"]
```

### Get a Detection Result

```python
import time

detection_uuid = "<detection_uuid>"
max_attempts = 10
wait_time_seconds = 3

for _ in range(max_attempts):
    response = Resemble.v2.deepfake_detection.get(detection_uuid)
    status = response.get("item", {}).get("status")

    if status in ("completed", "failed"):
        break

    time.sleep(wait_time_seconds)
```

### Create and Get a Detection (URL)

```python
import json

media_url = "https://example.com/sample-audio.wav"

response = Resemble.v2.deepfake_detection.detect_and_get(
    url=media_url,
    audio_source_tracing=True,
    intelligence=True,
    use_ood_detector=True,
)

# Full API response
print(json.dumps(response, indent=2))

# Parsed values
result = response.get("item", {})
status = response.get("item", {}).get("status")
```

### Create and Get a Detection (File Upload)

```python
import json

file_path = "path/to/local-audio.wav"

with open(file_path, "rb") as file:
    response = Resemble.v2.deepfake_detection.detect_and_get(
        file=file,
        audio_source_tracing=True,
        intelligence=True,
        use_ood_detector=True,
    )

# Full API response
print(json.dumps(response, indent=2))

# Parsed values

```
